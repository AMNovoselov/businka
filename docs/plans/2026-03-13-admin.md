# Admin Panel Implementation Plan

> **For Claude:** REQUIRED SUB-SKILL: Use superpowers:executing-plans to implement this plan task-by-task.

**Goal:** Add a `content.json` data layer and a GitHub API–powered admin panel so the photographer can edit texts and manage photos without touching code.

**Architecture:** Extract all editable content from `index.html` into `content.json`. `index.html` loads it via `fetch` on start. A hidden `admin-[hash].html` page reads and writes `content.json` + uploads photos to `photos/` — all via the GitHub Contents API using a Personal Access Token stored in `localStorage`.

**Tech Stack:** Plain HTML/CSS/JS, GitHub REST API v3 (no SDK).

```html
<script>
  // ── Tabs ──
  const tabBtns = document.querySelectorAll('.tab-btn');
  const tabPanels = document.querySelectorAll('.tab-panel');
  tabBtns.forEach(btn => {
    btn.addEventListener('click', () => {
      tabBtns.forEach(b => b.classList.remove('active'));
      tabPanels.forEach(p => p.classList.remove('active'));
      btn.classList.add('active');
      document.getElementById('tab-' + btn.dataset.tab).classList.add('active');
    });
  });

  // ── Masonry builder (accepts array of full URLs) ──
  function buildMasonry(id, photos) {
    const grid = document.getElementById('masonry-' + id);
    photos.forEach((url, i) => {
      const item = document.createElement('div');
      item.className = 'masonry-item stagger-item';
      item.dataset.delay = i * 60;
      item.innerHTML = `
        <img src="${url}" alt="">
        <div class="masonry-overlay">
          <svg viewBox="0 0 24 24" fill="none" stroke="currentColor" stroke-width="1.5">
            <path d="M15 3h6v6M9 21H3v-6M21 3l-7 7M3 21l7-7"/>
          </svg>
        </div>`;
      grid.appendChild(item);
    });
  }

  // ── Animations (called after content is injected) ──
  function initAnimations() {
    document.querySelectorAll('.hero-animate').forEach((el, i) => {
      setTimeout(() => el.classList.add('visible'), i * 150);
    });
    const observer = new IntersectionObserver((entries) => {
      entries.forEach(entry => {
        if (!entry.isIntersecting) return;
        const el = entry.target;
        const delay = el.dataset.delay ? parseInt(el.dataset.delay) : 0;
        setTimeout(() => el.classList.add('visible'), delay);
        observer.unobserve(el);
      });
    }, { threshold: 0.15 });
    document.querySelectorAll('.fade-up, .fade-left, .fade-right, .stagger-item').forEach(el => {
      observer.observe(el);
    });
  }

  // ── Hero parallax ──
  (function () {
    const hero = document.querySelector('.hero');
    const cards = document.querySelectorAll('.hero-photo-card');
    const depths = [0.035, 0.02, 0.015];
    hero.addEventListener('mousemove', (e) => {
      const cx = hero.offsetWidth / 2;
      const cy = hero.offsetHeight / 2;
      const dx = e.clientX - cx;
      const dy = e.clientY - cy;
      cards.forEach((card, i) => {
        const x = dx * depths[i];
        const y = dy * depths[i];
        card.style.transform = card.style.transform.replace(/\s*translate\([^)]*\)/g, '') + ` translate(${x}px, ${y}px)`;
      });
    });
    hero.addEventListener('mouseleave', () => {
      cards.forEach(card => {
        card.style.transform = card.style.transform.replace(/\s*translate\([^)]*\)/g, '');
      });
    });
  })();

  // ── Content initialiser ──
  function initContent(data) {
    // Text
    document.getElementById('js-tagline').textContent        = data.tagline;
    document.getElementById('js-hero-title').innerHTML       = data.hero_title_html;
    document.getElementById('js-hero-sub').textContent       = data.hero_subtitle;
    document.getElementById('js-about-name').textContent     = 'Привет, я ' + data.about_name;

    // Stats
    data.stats.forEach((s, i) => {
      const v = document.getElementById('js-stat-' + i + '-val');
      const l = document.getElementById('js-stat-' + i + '-lbl');
      if (v) v.textContent = s.value;
      if (l) l.textContent = s.label;
    });

    // Bio
    data.about_bio.forEach((text, i) => {
      const el = document.getElementById('js-bio-' + i);
      if (el) el.textContent = text;
    });

    // Photos
    data.hero_photos.forEach((url, i) => {
      const img = document.getElementById('js-hero-img-' + i);
      if (img) img.src = url;
    });
    const aboutImg = document.getElementById('js-about-img');
    if (aboutImg) aboutImg.src = data.about_photo;

    // Contacts
    const tg    = document.getElementById('js-contact-tg');
    const ig    = document.getElementById('js-contact-ig');
    const phone = document.getElementById('js-contact-phone');
    const lbl   = document.getElementById('js-phone-display');
    if (tg)    tg.href    = data.contacts.telegram_url;
    if (ig)    ig.href    = data.contacts.instagram_url;
    if (phone) phone.href = data.contacts.phone_url;
    if (lbl)   lbl.textContent = data.contacts.phone_display;

    // Portfolio
    buildMasonry('people', data.portfolio.people);
    buildMasonry('dogs',   data.portfolio.dogs);
    buildMasonry('events', data.portfolio.events);

    // Start animations after content is ready
    initAnimations();
  }

  // ── Bootstrap ──
  fetch('content.json')
    .then(r => { if (!r.ok) throw new Error(r.status); return r.json(); })
    .then(initContent)
    .catch(err => console.error('content.json load failed:', err));
</script>
```

  // ── Utilities ──────────────────────────────────────────────

  async function sha256(str) {
    const buf = await crypto.subtle.digest('SHA-256', new TextEncoder().encode(str));
    return Array.from(new Uint8Array(buf)).map(b => b.toString(16).padStart(2, '0')).join('');
  }

  function toB64(str) {
    return btoa(unescape(encodeURIComponent(str)));
  }
  function fromB64(b64) {
    return decodeURIComponent(escape(atob(b64.replace(/\s/g, ''))));
  }

  // ── GitHub API ─────────────────────────────────────────────

  async function ghGet(path) {
    const r = await fetch(GH_API + path, {
      headers: {
        'Authorization': 'Bearer ' + token,
        'Accept': 'application/vnd.github+json',
        'X-GitHub-Api-Version': '2022-11-28'
      }
    });
    if (!r.ok) throw new Error('GitHub ' + r.status + ': ' + await r.text());
    return r.json();
  }

  async function ghPut(path, body) {
    const r = await fetch(GH_API + path, {
      method: 'PUT',
      headers: {
        'Authorization': 'Bearer ' + token,
        'Accept': 'application/vnd.github+json',
        'Content-Type': 'application/json',
        'X-GitHub-Api-Version': '2022-11-28'
      },
      body: JSON.stringify(body)
    });
    if (!r.ok) throw new Error('GitHub ' + r.status + ': ' + await r.text());
    return r.json();
  }

  function compressImage(file, maxWidth = 1200) {
    return new Promise(resolve => {
      const img = new Image();
      const url = URL.createObjectURL(file);
      img.onload = () => {
        const ratio   = Math.min(1, maxWidth / img.width);
        const canvas  = document.createElement('canvas');
        canvas.width  = Math.round(img.width  * ratio);
        canvas.height = Math.round(img.height * ratio);
        canvas.getContext('2d').drawImage(img, 0, 0, canvas.width, canvas.height);
        URL.revokeObjectURL(url);
        canvas.toBlob(blob => resolve(blob), 'image/jpeg', 0.85);
      };
      img.src = url;
    });
  }

  function blobToBase64(blob) {
    return new Promise(resolve => {
      const reader = new FileReader();
      reader.onload  = () => resolve(reader.result.split(',')[1]);
      reader.readAsDataURL(blob);
    });
  }

        message: 'photo: upload ' + filename,

      } else {

      }
      renderPhotoGrid(cat);
      status.textContent = '✓ ' + filename + ' загружен. Нажмите «Сохранить» чтобы добавить на сайт.';
      setTimeout(() => { status.textContent = ''; }, 5000);
    } catch (e) {
      status.textContent = '✗ Ошибка: ' + e.message;
    }
  }

## Notes

- **Token scope:** Create a fine-grained PAT at https://github.com/settings/personal-access-tokens/new with "Contents: Read and Write" for the `businka` repo only. Do not use a classic token with full `repo` scope.
- **Photo size:** Files are compressed client-side to max 1200px wide, ~85% JPEG quality (~200–500 KB). Original file is never sent.
- **Concurrent edits:** If two people save at the same time, the second save will fail with a SHA conflict. Refresh the page and try again.
- **`about_photo` field:** Accepts any URL (Unsplash, GitHub Pages path, or relative path after upload).
- **Admin URL:** Share only the full URL including the hash. The page is not linked from the main site.
